export class TabViewDemo {

    msgs = [];

    }
}

使用

Tabview元素由一个或多个p-tabPanel元素组成。标签的标题是使用header属性定义的。


<p-tabView>
    <p-tabPanel header="Header 1">
        Content 1
    </p-tabPanel>
    <p-tabPanel header="Header 2">
        Content 2
    </p-tabPanel>
    <p-tabPanel header="Header 3">
        Content 3    
    </p-tabPanel>
</p-tabView>

选择

默认情况下选择第一个选项卡，以便在另一个面板上自定义此启用选定的属性。


<p-tabView>
    <p-tabPanel header="Header 1">
        Content 1
    </p-tabPanel>
    <p-tabPanel header="Header 2" [selected]="true">
        Content 2
    </p-tabPanel>
    <p-tabPanel header="Header 3">
        Content 3    
    </p-tabPanel>
</p-tabView>

可关闭的

启用closable属性时，会显示一个图标来关闭该选项卡。 当关闭选项卡时，onClose事件触发。


<p-tabPanel header="Header Text" [closable]="true">
    Content 1
</p-tabPanel>

如果要控制是否可以关闭选项卡，请启用controlClose属性，以使控件能够在onClose事件中决定。 在这种情况下，事件会得到一个称为close callback的第三个参数，并调用该函数实际上关闭该选项卡。 


<p-tabView [controlClose]="true" (onClose)="handleClose($event)">
<p-tabPanel header="Header Text" [closable]="true">
    Content 1
</p-tabPanel>
</p-tabView>



handleClose(e) {
   if(condition)
       e.close();
}

不可用

可以禁用选项卡，通过在面板上设置disabled属性来禁用内容。


<p-tabPanel header="Header Text" [disabled]="true">
    Content 1
</p-tabPanel>

图标

可以使用tabPanel的leftIcon和rightIcon属性将标题放置在标题的左侧和右侧。


<p-tabPanel header="Header Text" leftIcon="fa-bell-o" rightIcon="fa-bookmark-o">;
    Content
</p-tabPanel>

布局

TabView有四个不同的方向： "top", "bottom", "left"和"right"。 Top是默认值，可以使用orientation属性更改。


<p-tabView orientation="left">

程序化控制

可以使用activeIndex属性以编程方式控制选项卡，该属性定义活动选项卡的索引。

HTML代码示例

<button type="button" pButton icon="fa fa-chevron-up" (click)="openPrev()"></button>
<button type="button" pButton icon="fa fa-chevron-down" (click)="openNext()"></button>

<p-tabView [activeIndex]="index">
    <p-tabPanel header="Header 1">
        Content 1
    </p-tabPanel>
    <p-tabPanel header="Header 2">
        Content 2
    </p-tabPanel>
    <p-tabPanel header="Header 3">
        Content 3    
    </p-tabPanel>
</p-tabView>
TS代码示例

export class TabViewDemo {

    index: number = 0;

    openNext() {
        this.index = (this.index === 2) ? 0 : this.index + 1;
    }

    openPrev() {
        this.index = (this.index === 0) ? 2 : this.index - 1;
    }
    
}

懒加载

只有通过初始化活动选项卡，延迟加载才有助于初始加载性能，未激活的选项卡在被选中之前不会初始化。默认情况下缓存加载的标签页内容，以便在重新选择时不会再次创建。您可以使用TabPanel上的缓存属性来配置此行为。 当具有pTemplate =“content”的ngTemplate时，TabPanel被指定为懒惰。


<p-tabView>
    <p-tabPanel header="Header 1">
        Content 1
    </p-tabPanel>
    <p-tabPanel header="Header 2" [selected]="true">
        <ng-template pTemplate="content">
            Complex Content to Lazy Load
        </ng-template>
    </p-tabPanel>
    <p-tabPanel header="Header 3">
        <ng-template pTemplate="content">
            Complex Content to Lazy Load
        </ng-template>    
    </p-tabPanel>
</p-tabView>
